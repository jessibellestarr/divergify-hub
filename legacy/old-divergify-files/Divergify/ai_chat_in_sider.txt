  static createOpenLoop(task, steps) {

  }

  static generateHint(nextStep) {
    // Show just enough to create curiosity
    const words = nextStep.split(' ');
    if (words.length <= 3) return `${words[0]}...`;
    return `${words[0]} ${words[1]}...`;
  }

  static completeStep(openLoop, stepIndex) {
    // Mark step complete, reveal next
    const updated = { ...openLoop };
    updated.steps[stepIndex].completed = true;
    updated.completedSteps += 1;

    // Reveal next step (if exists)
    if (stepIndex + 1 < updated.steps.length) {
      updated.steps[stepIndex + 1].visible = true;
      updated.revealedSteps += 1;

      // Show hint for step AFTER next
      if (stepIndex + 2 < updated.steps.length) {
        updated.steps[stepIndex + 1].hint = this.generateHint(
          updated.steps[stepIndex + 2].text
        );
      }
    }

    return updated;
  }

  static getProgressMessage(openLoop, persona) {
    const remaining = openLoop.totalSteps - openLoop.completedSteps;
    
    const messages = {
      takoda: `${remaining} left. You're on a roll.`,
      coach: `${remaining} more reps. Keep pushing.`,
      strategist: `${openLoop.completedSteps}/${openLoop.totalSteps} executed. Maintain pace.`,
      soothe: `You're doing beautifully. ${remaining} gentle steps remain.`,
      gremlin: `Only ${remaining} more to unlock the chaos. üòà`,
    };

    return messages[persona] || messages.takoda;
  }

  static shouldRevealNextStep(openLoop) {
    // Prevent user from seeing ALL steps (ruins open loop)
    const lastVisible = openLoop.steps.findIndex(s => s.visible && !s.completed);
    return lastVisible >= 0 && lastVisible < openLoop.steps.length - 1;
  }
}

// CORE ENGINE: FractuationEngine.js
// Task Breakdown into 2-Minute Chunks
// ===================================

export class FractuationEngine {
  static fractuate(task) {
    // Break task into micro-steps (2-5 min each)
    // This is where AI/GPT would help, but for now, rule-based

    if (task.steps && task.steps.length > 0) {
      return task.steps; // Already fractuated
    }

    // Default fractuation patterns
    const patterns = {
      email: [
        'Open email app',
        'Start new message',
        'Write terrible first sentence',
        'Write 2 more sentences',
        'Add recipient & hit send',
      ],
      writing: [
        'Open document',
        'Write one bad sentence',
        'Write 3 more bad sentences',
        'Add one detail or example',
        'Save and close',
      ],
      research: [
        'Open browser',
        'Type search query',
        'Skim first 3 results',
        'Copy one useful quote',
        'Paste in notes',
      ],
      meeting: [
        'Open calendar',
        'Check meeting time',
        'Prepare one question',
        'Join 2 minutes early',
        'Take one note',
      ],
      generic: [
        'Define the tiniest first action',
        'Do that action',
        'Define the next tiny action',
        'Do that action',
        'Celebrate shitty progress',
      ],
    };

    // Simple keyword matching (replace with AI later)
    const taskLower = task.text.toLowerCase();
    
    if (taskLower.includes('email') || taskLower.includes('message')) {
      return patterns.email;
    }
    if (taskLower.includes('write') || taskLower.includes('blog') || taskLower.includes('post')) {
      return patterns.writing;
    }
    if (taskLower.includes('research') || taskLower.includes('find') || taskLower.includes('look up')) {
      return patterns.research;
    }
    if (taskLower.includes('meeting') || taskLower.includes('call')) {
      return patterns.meeting;
    }

    return patterns.generic;
  }

  static estimateEnergyCost(step) {
    // Energy levels: low, medium, high
    const keywords = {
      low: ['open', 'check', 'skim', 'copy', 'paste', 'save'],
      medium: ['write', 'draft', 'prepare', 'organize', 'respond'],
      high: ['create', 'design', 'solve', 'analyze', 'present'],
    };

    const stepLower = step.toLowerCase();

    if (keywords.low.some(k => stepLower.includes(k))) return 'low';
    if (keywords.high.some(k => stepLower.includes(k))) return 'high';
    return 'medium';
  }

  static addEmotionalAnchors(task) {
    // Questions to create F.E.A.R. profile
    return {
      whyQuestion: "Why does this matter to you?",
      consequenceQuestion: "What happens if you DON'T do this?",
      rewardQuestion: "How will you feel when it's done?",
    };
  }
}

export class BaselineEngine {
  // Track historical task completion times
  static async getBaseline(taskType) {
    try {
      const history = await AsyncStorage.getItem(`baseline_${taskType}`);
      if (!history) return null;

      const times = JSON.parse(history);
      const avg = times.reduce((a, b) => a + b, 0) / times.length;
      return Math.round(avg); // seconds
    } catch (e) {
      return null;
    }
  }

  static async recordTime(taskType, seconds) {
    try {
      const history = await AsyncStorage.getItem(`baseline_${taskType}`);
      const times = history ? JSON.parse(history) : [];
      
      // Keep last 10 attempts
      times.push(seconds);
      if (times.length > 10) times.shift();

      await AsyncStorage.setItem(`baseline_${taskType}`, JSON.stringify(times));
      
      return this.calculateImprovement(times);
    } catch (e) {
      console.error('Error recording baseline:', e);
      return null;
    }
  }

  static calculateImprovement(times) {
    if (times.length < 2) return null;

    const previous = times[times.length - 2];
    const current = times[times.length - 1];
    const diff = previous - current;
    const percentChange = (diff / previous) * 100;

    return {
      improved: diff > 0,
      seconds: Math.abs(diff),
      percent: Math.abs(percentChange).toFixed(1),
    };
  }

  static getMotivationalMessage(improvement, persona) {
    if (!improvement) return null;

    const messages = {
      takoda: improvement.improved 
        ? `${improvement.seconds}s faster. You're getting scary good.`
        : `${improvement.seconds}s slower. Distracted or just enjoying it?`,
      coach: improvement.improved
        ? `NEW RECORD! ${improvement.percent}% improvement! üî•`
        : `Slower today, but you SHOWED UP. That's the win.`,
      strategist: improvement.improved
        ? `Efficiency increased ${improvement.percent}%. Optimal trajectory.`
        : `Performance variance detected. Analyze and adjust.`,
      soothe: improvement.improved
        ? `You're growing. ${improvement.seconds} seconds faster. Be proud.`
        : `You finished. That's what matters. Time isn't everything.`,
      gremlin: improvement.improved
        ? `HAH! ${improvement.seconds}s faster! You're becoming a MACHINE.`
        : `Slower? Must've been daydreaming about chaos. üòà`,
    };

    return messages[persona] || messages.takoda;
  }

  static estimateTaskDuration(taskType, baseline) {
    // Provide time estimate with buffer
    if (!baseline) return 'Unknown (first attempt)';

    const buffer = Math.round(baseline * 0.2); // 20% buffer
    const estimate = baseline + buffer;

    const minutes = Math.floor(estimate / 60);
    const seconds = estimate % 60;

    if (minutes > 0) {
      return `~${minutes}m ${seconds}s (your baseline: ${Math.floor(baseline / 60)}m)`;
    }
    return `~${estimate}s (your baseline: ${baseline}s)`;
  }
}

import * as Notifications from 'expo-notifications';

export class NotificationEngine {
  static async requestPermissions() {
    const { status } = await Notifications.requestPermissionsAsync();
    return status === 'granted';
  }

  static async scheduleHypnoticReminder(task, fearProfile, persona, messageIndex) {
    const hasPermission = await this.requestPermissions();
    if (!hasPermission) return;

    const message = FEAREngine.getHypnoticMessage(fearProfile, persona, messageIndex);
    const interval = fearProfile.repetition.interval * 60; // Convert to seconds

    await Notifications.scheduleNotificationAsync({
      content: {
        title: PersonaSystem.getPersona(persona).emoji,
        body: message,
        sound: true,
        priority: 'high',
        data: { taskId: task.id },
      },
      trigger: {
        seconds: interval,
        repeats: false,
      },
    });

    // Update message count
    fearProfile.repetition.messagesSent += 1;
  }

  static async scheduleHabitTrance(habit, persona) {
    const hasPermission = await this.requestPermissions();
    if (!hasPermission) return;

    // Daily notification at set time
    const [hour, minute] = habit.time.split(':').map(Number);

    const message = this.getHabitTranceCopy(habit, persona);

    await Notifications.scheduleNotificationAsync({
      content: {
        title: `${PersonaSystem.getPersona(persona).emoji} Daily Trance`,
        body: message,
        sound: true,
        priority: 'high',
      },
      trigger: {
        hour,
        minute,
        repeats: true,
      },
    });
  }

  static getHabitTranceCopy(habit, persona) {
    const templates = {
      takoda: `You're the kind of person who ${habit.name}. You've done it before. Today's the day you already started.`,
      coach: `${habit.name.toUpperCase()}. That's who you are. That's what you do. LET'S GO.`,
      strategist: `Habit execution window open. ${habit.name}. Consistency compounds.`,
      soothe: `Gentle reminder: ${habit.name}. You're building something beautiful, one day at a time.`,
      gremlin: `Time to ${habit.name}. Or don't. But you know you want to. üòà`,
    };

    return templates[persona] || templates.takoda;
  }

  static async cancelAll() {
    await Notifications.cancelAllScheduledNotificationsAsync();
  }
}

import React, { useState, useEffect } from 'react';
import {
  View,
  Text,
  ScrollView,
  StyleSheet,
} from 'react-native';
import AsyncStorage from '@react-native-async-storage/async-storage';

export default function StatsScreen() {
  const [stats, setStats] = useState({
    tasksCompleted: 0,
    currentStreak: 0,
    longestStreak: 0,
    totalFocusTime: 0,
    habitsActive: 0,
  });

  useEffect(() => {
    calculateStats();
  }, []);

  const calculateStats = async () => {
    try {
      // Get tasks
      const tasksData = await AsyncStorage.getItem('divergify_tasks');
      const tasks = tasksData ? JSON.parse(tasksData) : [];
      const completed = tasks.filter(t => 
        t.openLoop.completedSteps === t.openLoop.totalSteps
      ).length;

      // Get habits
      const habitsData = await AsyncStorage.getItem('divergify_habits');
      const habits = habitsData ? JSON.parse(habitsData) : [];

      // Calculate streaks
      const streaks = habits.map(h => h.streak);
      const currentStreak = Math.max(...streaks, 0);
      const longestStreak = currentStreak; // Simplified

      // Get focus time (from baseline data)
      // This would sum all recorded session times
      const totalFocusTime = 0; // Implement based on stored data

      setStats({
        tasksCompleted: completed,
        currentStreak,
        longestStreak,
        totalFocusTime,
        habitsActive: habits.length,
      });
    } catch (e) {
      console.error('Error calculating stats:', e);
    }
  };

  const formatFocusTime = (seconds) => {
    const hours = Math.floor(seconds / 3600);
    const minutes = Math.floor((seconds % 3600) / 60);
    return `${hours}h ${minutes}m`;
  };

  return (
    <ScrollView style={styles.container}>
      <View style={styles.header}>
        <Text style={styles.title}>üìä Your Stats</Text>
        <Text style={styles.subtitle}>Progress without pressure</Text>

          <Text style={styles.statValue}>{stats.tasksCompleted}</Text>
          <Text style={styles.statLabel}>Tasks Completed</Text>
        </View>

        <View style={[styles.statCard, { borderLeftColor: '#2FB3FF' }]}>
          <Text style={styles.statValue}>{stats.currentStreak}</Text>
          <Text style={styles.statLabel}>Current Streak</Text>
        </View>

        <View style={[styles.statCard, { borderLeftColor: '#FFD166' }]}>
          <Text style={styles.statValue}>{stats.longestStreak}</Text>
          <Text style={styles.statLabel}>Longest Streak</Text>
        </View>

        <View style={[styles.statCard, { borderLeftColor: '#8A7CFF' }]}>
          <Text style={styles.statValue}>{stats.habitsActive}</Text>
          <Text style={styles.statLabel}>Active Habits</Text>
        </View>

        <View style={[styles.statCard, { borderLeftColor: '#FF6B6B' }]}>
          <Text style={styles.statValue}>

            document.getElementById('streakCount').textContent = stats.streak;

        // Add task
        function addTask() {
            const input = document.getElementById('taskInput');
            const text = input.value.trim();
            if (!text) return;

            tasks.push({
                id: Date.now(),
                text: text,
                done: false,
                created: new Date().toISOString()
            });

            stats.total++;
            input.value = '';
            renderTasks();
            updateVoiceDisplay();
            saveData();
        }

        // Toggle task
        function toggleTask(id) {
            const task = tasks.find(t => t.id === id);
            if (!task) return;

            task.done = !task.done;
            
            if (task.done) {
                stats.completed++;
                stats.streak++;
                showReward();
            } else {
                stats.completed--;
                stats.streak = Math.max(0, stats.streak - 1);
            }

            renderTasks();
            updateVoiceDisplay();
            saveData();
        }

        // Delete task
        function deleteTask(id) {
            const task = tasks.find(t => t.id === id);
            if (task && task.done) {
                stats.completed--;
            }
            stats.total--;
            tasks = tasks.filter(t => t.id !== id);
            renderTasks();
            updateVoiceDisplay();
            saveData();
        }

            const rewards = voices[currentVoice].rewards;

            const input = document.getElementById('taskInput');

        // Toggle task
        function toggleTask(id) {
            const task = tasks.find(t => t.id === id);
            if (!task) return;

            task.done = !task.done;
            
            if (task.done) {
                stats.completed++;
                stats.streak++;
                showReward();
            } else {
                stats.completed--;
                stats.streak = Math.max(0, stats.streak - 1);

            renderTasks();
            updateVoiceDisplay();
            saveData();

            updateVoiceDisplay();
            saveData();
