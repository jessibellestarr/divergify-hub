<script>

// ===== STATE & STORAGE =====

let tasks = JSON.parse(localStorage.getItem('divergify_tasks')) || [];

let recognition = null;

let isListening = false;


// ===== VOICE RECOGNITION SETUP =====

function initVoiceRecognition() {

if (!('webkitSpeechRecognition' in window) && !('SpeechRecognition' in window)) {

console.warn('Speech recognition not supported');

return null;

}


const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition;

const recog = new SpeechRecognition();

recog.continuous = false;

recog.interimResults = true;

recog.lang = 'en-US';


recog.onstart = () => {

isListening = true;

document.getElementById('voiceBtn').classList.add('listening');

document.getElementById('captureBox').classList.add('active');

document.getElementById('voiceStatus').textContent = 'üéôÔ∏è Listening...';

};


recog.onresult = (event) => {

const transcript = Array.from(event.results)

.map(result => result[0].transcript)

.join('');

document.getElementById('taskInput').value = transcript;

};


recog.onend = () => {

isListening = false;

document.getElementById('voiceBtn').classList.remove('listening');

document.getElementById('captureBox').classList.remove('active');

document.getElementById('voiceStatus').textContent = '';

};


recog.onerror = (event) => {

console.error('Speech recognition error:', event.error);

document.getElementById('voiceStatus').textContent = '‚ö†Ô∏è Mic error. Try again.';

isListening = false;

document.getElementById('voiceBtn').classList.remove('listening');

document.getElementById('captureBox').classList.remove('active');

};


return recog;

}


recognition = initVoiceRecognition();


// ===== VOICE BUTTON =====

document.getElementById('voiceBtn').addEventListener('click', () => {

if (!recognition) {

alert('Voice input not supported in this browser. Try Chrome, Edge, or Safari.');

return;

}


if (isListening) {

recognition.stop();

} else {

recognition.start();

}

});


// ===== ADD TASK =====

function addTask() {

const input = document.getElementById('taskInput');

const text = input.value.trim();

if (!text) return;


const task = {

id: Date.now(),

text: text,

timestamp: new Date().toLocaleString()

};


tasks.unshift(task);

saveTasks();

renderTasks();

input.value = '';

input.focus();

}


document.getElementById('addBtn').addEventListener('click', addTask);

document.getElementById('taskInput').addEventListener('keypress', (e) => {

if (e.key === 'Enter') addTask();

});


// ===== DELETE TASK =====

function deleteTask(id) {

tasks = tasks.filter(task => task.id !== id);

saveTasks();

renderTasks();

}


// ===== CLEAR ALL =====

document.getElementById('clearBtn').addEventListener('click', () => {

if (tasks.length === 0) return;

if (confirm('Clear all tasks? This cannot be undone.')) {

tasks = [];

saveTasks();

renderTasks();

}

});


// ===== SAVE TO LOCALSTORAGE =====

function saveTasks() {

localStorage.setItem('divergify_tasks', JSON.stringify(tasks));

}


// ===== RENDER TASKS =====

function renderTasks() {

const container = document.getElementById('tasksList');

if (tasks.length === 0) {

container.innerHTML = `

<div class="empty-state">

<p>‚ú® No captures yet</p>

<p style="font-size: 0.9rem;">Start brain dumping above ‚Üë</p>

</div>

`;

return;

}


container.innerHTML = tasks.map(task => `

<div class="task-item">

<div class="task-content">

<div class="task-text">${escapeHtml(task.text)}</div>

<div class="task-time">${task.timestamp}</div>

</div>

<button class="delete-btn" onclick="deleteTask(${task.id})">Delete</button>

</div>

`).join('');

}


// ===== ESCAPE HTML =====

function escapeHtml(text) {

const div = document.createElement('div');

div.textContent = text;

return div.innerHTML;

}


// ===== INITIAL RENDER =====

renderTasks();

</script>
